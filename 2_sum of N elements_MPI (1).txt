Steps To Run on Ubuntu:

Step 1: Install MPI
                 sudo apt update
                 sudo apt install mpich

                 mpicc --version

Step 2: Create File
             mpi_sum.c

Step 3: Compile
        mpicc mpi_sum.c -o mpi_sum

Step 4: Run Program
        mpirun -np 4 ./mpi_sum




mpi_sum.c

#include <stdio.h>
#include <mpi.h>

int main(int argc, char *argv[])
{
    int rank, size;
    int arr[100], n;
    int local_sum = 0, total_sum = 0;

    MPI_Init(&argc, &argv);

    MPI_Comm_rank(MPI_COMM_WORLD, &rank);
    MPI_Comm_size(MPI_COMM_WORLD, &size);

    // Only process 0 takes input
    if(rank == 0)
    {
        printf("Enter number of elements: ");
        fflush(stdout);

        scanf("%d", &n);

        printf("Enter array elements:\n");
        fflush(stdout);

        for(int i = 0; i < n; i++)
        {
            scanf("%d", &arr[i]);
        }
    }

    // Broadcast data to all processors
    MPI_Bcast(&n, 1, MPI_INT, 0, MPI_COMM_WORLD);
    MPI_Bcast(arr, n, MPI_INT, 0, MPI_COMM_WORLD);

    int elements_per_process = n / size;

    int start = rank * elements_per_process;

    int end;

    if(rank == size - 1)
        end = n;
    else
        end = start + elements_per_process;

    // Calculate local sum
    for(int i = start; i < end; i++)
    {
        local_sum += arr[i];
    }

    printf("Processor %d calculated intermediate sum = %d\n",
            rank, local_sum);

    // Calculate final sum
    MPI_Reduce(&local_sum, &total_sum,
               1, MPI_INT, MPI_SUM,
               0, MPI_COMM_WORLD);

    // Print final sum
    if(rank == 0)
    {
        printf("Final Sum = %d\n", total_sum);
    }

    MPI_Finalize();

    return 0;
}
